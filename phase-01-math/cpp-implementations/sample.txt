#include "vector2d.h"
#include <iostream>
#include <iomanip>
#include <cmath>

// Constructor
Vector2D::Vector2D(double x, double y) : x(x), y(y) {}

// Vector addition
// Geometric: tip-to-tail method
// Numeric: add corresponding components
Vector2D Vector2D::add(const Vector2D& other) const {
    return Vector2D(x + other.x, y + other.y);
}

// Scalar multiplication
// Stretches/shrinks vector, flips direction if negative
Vector2D Vector2D::multiply(double scalar) const {
    return Vector2D(scalar * x, scalar * y);
}

// Display vector in [x, y] format
void Vector2D::print() const {
    std::cout << "[" << std::fixed << std::setprecision(2) 
              << x << ", " << y << "]";
}
